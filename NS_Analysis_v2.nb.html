p1 <- ggplot(T_n_Leadlag, aes(x = StrandStatus, y= percentage, fill = StrandStatus)) +
  geom_bar(
    stat = "identity", 
    color = "black",
    size = 0.25,
    width =0.8) +
  scale_y_continuous(
