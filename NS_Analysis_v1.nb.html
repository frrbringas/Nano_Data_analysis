Fork_nucleosomes <- BC01_Nucleosomes %>% 
  count(StrandDirection) %>% 
  mutate(percentage = n/T_n * 100) %>% 
  mutate(StrandDirection = factor(StrandDirection, levels = c("fwdLead", "revLag", "revLead", "fwdLag")))

#Ploting the barchart

ggplot(Fork_nucleosomes, aes(x = StrandDirection, y = percentage, fill = StrandDirection)) +
  geom_col(
    stat = "identity",
     color = "black",
    size = 0.25) +
  scale_y_continuous(
    breaks = seq(0, 500, by =10), 
    limits = c(0, 50), 
    expand = c(0,0)) +
  geom_text(
    aes(label = paste0(round(percentage, 1), "%")),
            vjust = -0.5) +
  labs(
    y= "% of Nucleosomes", 
    x = "Strands") +
  scale_x_discrete(
